// 核心设计原则：测试间完全隔离，不共享 QueryClient
export function createMockQueryClient() { return new QueryClient({ logger: { log: () => {}, warn: () => {}, // error: jest.fn() ← 避免 React Query 内部错误干扰测试 }, defaultOptions: { queries: { retry: false, // 防止测试中自动重试导致超时 staleTime: Infinity, // 防止自动 refetch 干扰断言 gcTime: Infinity, // 防止自动 GC networkMode: 'offline', // 禁用网络（用 MSW 接管） }, mutations: { retry: false, networkMode: 'offline', }, }, })
}
``` **关键点：**
- `new QueryClient()` 每测试创建一个新实例
- `staleTime: Infinity` — 防止测试期间"幽灵 refetch"导致断言失败
- `networkMode: 'offline'` + MSW — 完全控制网络层 --- #### ② `render-with-providers.tsx` — 所有组件测试的统一入口 ```tsx
// 设计原则：调用方只需要 render(<MyComponent />)，不需要知道内部依赖了哪些 provider
import { render, RenderOptions } from '@testing-library/react'
import { QueryClient, QueryClientProvider } from '@tanstack/react-query'
import { createMockQueryClient } from './mock-query-client'
import { StoreProvider } from '@/lib/store' // Store state 可以按测试灵活覆盖
export function renderWithProviders( ui: React.ReactElement, { preloadedState = {}, // 每个测试可自定义初始状态 storeOverrides = {}, // 每个测试可 mock 特定 store 函数 queryClientOverrides = {}, // 每个测试可自定义 QueryClient 选项 ...renderOptions }: RenderWithProvidersOptions = {}
) { const queryClient = new QueryClient({ logger: { log: ()=>{}, warn: ()=>{}, error: ()=>{} }, defaultOptions: { queries: { retry: false, staleTime: Infinity, gcTime: Infinity, networkMode: 'offline' }, mutations: { retry: false, networkMode: 'offline' }, ...queryClientOverrides, }, }) function AllProviders({ children }: { children: React.ReactNode }) { return ( <QueryClientProvider client={queryClient}> <StoreProvider initialState={preloadedState}> {children} </StoreProvider> </QueryClientProvider> ) } return render(ui, { wrapper: AllProviders, ...renderOptions })
} // 导出标准 RTL 工具（使用方不需要单独导入）
export * from '@testing-library/react'
``` **为什么这样封装？**
- `render(<UserCard />)` 就够了，不需要 `render(ui, { wrapper: AllProviders })`
- 每个测试的 `QueryClient` 是独立实例，互相不污染
- `preloadedState` 让每个测试精确控制 Zustand 初始状态 --- #### ③ `__mocks__/store.ts` — Zustand 全局自动拦截 ```ts
// 设计原则：测试 src/lib/store 的使用者时，自动替换为 mock
// Jest 自动模块系统会拦截所有 '@/lib/store' 和 '@/lib/store/user-store' 的导入 const mockUserStore = { state: { id: '1', name: 'Test User', role: 'user' as const }, setUser: jest.fn(), resetUser: jest.fn(),
} const mockUIStore = { state: { sidebarOpen: false, theme: 'light' as const }, toggleSidebar: jest.fn(), setTheme: jest.fn(),
} jest.mock('@/lib/store/user-store', () => ({ __esModule: true, useUserStore: () => mockUserStore,
})) jest.mock('@/lib/store/ui-store', () => ({ __esModule: true, useUIStore: () => mockUIStore,
})) // 导出工厂函数，允许单个测试覆盖
export function createMockUserStore(overrides = {}) { return { ...mockUserStore, ...overrides }
} export function createMockUIStore(overrides = {}) { return { ...mockUIStore, ...overrides }
}
``` **关键点：**
- `jest.mock` 在文件顶层注册，**Jest 自动对所有测试生效**
- 任何测试文件 `import { useUserStore } from '@/lib/store/user-store'` 都会拿到 mock
- 单个测试可以用 `jest.mocked(useUserStore).mockReturnValue(...)` 局部覆盖 --- ### 三、MSW（网络层 mock） ```ts

|---|------|------|
| 1 | **每个测试 `new QueryClient()`** | 防止多个测试共享缓存导致状态污染 |
| 2 | **`renderWithProviders` 是唯一入口** | 组件永远不知道自己在测试环境 |
| 3 | **先 render 再 act 再 assert** | `render()` 已经是 async（Suspense），不需要额外 waitFor |
| 4 | **MSW `networkMode: 'offline'`** | 确保测试不碰真实网络，只走 MSW |
| 5 | **store 用 `getState()` 测试逻辑** | Zustand 的 `getState()` 绕过了 React 渲染，直接测业务逻辑 |
| 6 | **Mock 文件放 `__mocks__/`** | Jest 自动拦截，不污染源代码 |
| 7 | **测试文件名：`*.test.tsx`** | 组件和 Hook；`*.test.ts` 纯逻辑和工具函数 |
| 8 | **setup.ts 只做全局清理** | 每个测试自己的 mock 在测试文件内写 |
| 9 | **不用 enzyme，只用 RTL** | 更接近真实用户交互，防止测试实现细节 |
| 10 | **Hook 测试用 `renderHook`** | from `@testing-library/react`，不用第三方 | --- **确认这个思路 OK 之后，我立刻开始生成代码。** 有没有需要调整的地方？比如：
