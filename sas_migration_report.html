SAS — RFM Scoring
/* SAS: RFM Score Calculation */ data work.rfm; set work.customers; /* Recency Score */ if recency_days <= 30 then r_score = 5; else if recency_days <= 90 then r_score = 4; else r_score = 1; rfm_score = cats(r_score, f_score, m_score); run;
Python — RFM Scoring
# Python: RFM Score Calculation import pandas as pd # Recency Score df['r_score'] = pd.cut( df['recency_days'], bins=[-1, 30, 90, 180, 365, float('inf')], labels=[5, 4, 3, 2, 1] ).astype(int) # RFM combined score df['rfm_score'] = ( df['r_score'].astype(str) + df['f_score'].astype(str) + df['m_score'].astype(str) )
SAS — PROC SQL Aggregation
/* SAS: Segment Summary */ proc sql; create table work.summary as select segment, count(*) as customer_count, sum(monetary) as total_revenue, mean(monetary) as avg_revenue from work.customers group by segment order by total_revenue desc; quit;
Python — pandas groupby
# Python: Segment Summary summary = df.groupby('segment').agg( customer_count=('customer_id', 'count'), total_revenue=('monetary', 'sum'), avg_revenue=('monetary', 'mean') ).reset_index() summary = summary.sort_values( 'total_revenue', ascending=False )
